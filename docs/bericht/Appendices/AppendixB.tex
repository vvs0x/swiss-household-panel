  foreign::read.spss(
    file = path,
    to.data.frame = TRUE,

    use.missings = FALSE
  ) |>

\section{Konstruktion der Variable NAT}

\begin{lstlisting}[language=R, caption=Verdichtung der Nationalitätsvariablen]
is_swiss <- function(x) x %in% c(8100, 756)

panel_adults <- panel_adults |>
  mutate(
    n_ch = as.integer(replace_na(is_swiss(nat_1_), FALSE)) +
           as.integer(replace_na(is_swiss(nat_2_), FALSE)) +
           as.integer(replace_na(is_swiss(nat_3_), FALSE)),
    n_nat = as.integer(!is.na(nat_1_)) +
            as.integer(!is.na(nat_2_)) +
            as.integer(!is.na(nat_3_)),
    NAT = case_when(
      n_ch >= 1 & n_nat == n_ch ~ "CH",
      n_ch >= 1 & n_nat > n_ch  ~ "CH+Andere",
      n_ch == 0 & n_nat >= 1    ~ "Andere",
      TRUE ~ NA_character_
    )
  )
\end{lstlisting}
